49void bno055_writeData(uint8_t reg, uint8_t data) {
50 uint8_t txdata[2] = {reg, data};
51 uint8_t status;
52 status = HAL_I2C_Master_Transmit(_bno055_i2c_port, BNO055_I2C_ADDR << 1,
53 txdata, sizeof(txdata), 10);
54 if (status == HAL_OK) {
55 return;
56 }
57
58 if (status == HAL_ERROR) {
59 printf("HAL_I2C_Master_Transmit HAL_ERROR\r\n");
60 } else if (status == HAL_TIMEOUT) {
61 printf("HAL_I2C_Master_Transmit HAL_TIMEOUT\r\n");
62 } else if (status == HAL_BUSY) {
63 printf("HAL_I2C_Master_Transmit HAL_BUSY\r\n");
64 } else {
65 printf("Unknown status data %d", status);
66 }
67
68 uint32_t error = HAL_I2C_GetError(_bno055_i2c_port);
69 if (error == HAL_I2C_ERROR_NONE) {
70 return;
71 } else if (error == HAL_I2C_ERROR_BERR) {
72 printf("HAL_I2C_ERROR_BERR\r\n");
73 } else if (error == HAL_I2C_ERROR_ARLO) {
74 printf("HAL_I2C_ERROR_ARLO\r\n");
75 } else if (error == HAL_I2C_ERROR_AF) {

80 printf("HAL_I2C_ERROR_DMA\r\n");
81 } else if (error == HAL_I2C_ERROR_TIMEOUT) {
82 printf("HAL_I2C_ERROR_TIMEOUT\r\n");
83 }
84
85 HAL_I2C_StateTypeDef state = HAL_I2C_GetState(_bno055_i2c_port);
86 if (state == HAL_I2C_STATE_RESET) {
87 printf("HAL_I2C_STATE_RESET\r\n");
88 } else if (state == HAL_I2C_STATE_READY) {

95 printf("HAL_I2C_STATE_BUSY_RX\r\n");
96 } else if (state == HAL_I2C_STATE_LISTEN) {
97 printf("HAL_I2C_STATE_LISTEN\r\n");
98 } else if (state == HAL_I2C_STATE_BUSY_TX_LISTEN) {
99 printf("HAL_I2C_STATE_BUSY_TX_LISTEN\r\n");
